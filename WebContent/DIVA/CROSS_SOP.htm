2          執行以下指令(注意:黃底跟綠底的螢光筆要改成匯入DB時的TABLE NAME):

 

DECLARE @cols AS NVARCHAR(MAX),

    @query  AS NVARCHAR(MAX)

 

select @cols = STUFF((SELECT ',' + QUOTENAME([soc-distno])

                    from [O162-work detail]

                    group by [soc-distno]

                    order by [soc-distno]

            FOR XML PATH(''), TYPE

            ).value('.', 'NVARCHAR(MAX)')

        ,1,1,'')

 

set @query = 'select [Dist No] as Expr1 ,[Account Id] AS Expr2 ,[chinese name] AS Expr3 ,[合計Royalties],' + @cols + ' from  (

                select a.[Dist No] ,a.[Account Id],sum(a.[Royalties]) as Royalties,a.[soc-distno],b.[合計Royalties],c.[chinese name]

                from [O162-work detail] a

                join (select [Account Id],sum([Royalties]) as [合計Royalties] from [O162-work detail] group by [Account Id]) b on b.[Account Id] = a.[Account Id]

               left join [O162-retain] c on a.[Account Id] = c.[ip base no]

                group by a.[Dist No] ,a.[Account Id],a.[soc-distno] ,b.[合計Royalties],c.[chinese name]

            ) x

            pivot

            (

                sum(Royalties)

                for [soc-distno] in (' + @cols + ')

            ) p

            order by [Account Id]

            '

           

execute(@query);
